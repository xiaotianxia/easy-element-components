## 示例
```html
<imp-suggest
    v-model="queryparam.agentUserId"
    :dao="$dao"
    :extro-param="{ limitSize: 20 }"
    :placeholder="`输入代理商${type === 'name' ? '' : '公司'}名称搜索...`"
    :query-param="type"
    api-path="activity.getAgentUser"
    res-path="data"
    unique-key="userId"
    label="${companyName}_${userName}"
    @change="queryparam.agentUserName = $event.userName;"
>
</imp-suggest>
```
